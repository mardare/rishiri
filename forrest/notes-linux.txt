KEYS/
  CTRL+ALT+T  new terminal

  CTRL+ALT+F1 tty1...
  ALT+F7      gui tty

CONFIG/

  change tty(console) font
  sudo dpkg-reconfigure console-setup

  #enable root su
  sudo usermod root -p <password>
  sudo passwd root
  su
  
  #vi fix arrow-keys issue
  vi ~/.exrc
    add "set nocompatible"  
  #vi fix arrow-keys issue for root
  sudo -i touch /root/.exrc
  sudo -i vi /root/.exrc
    add "set nocompatible"  

INSTALL/
  
  list:
   dpkg -l|grep -i "java"
  
  uninstall:
   sudo apt-get purge openjdk-\* icedtea-\* icedtea6-\*
   which java
   which javac

CMD/

  tar zxvf jre-8u45-linux-x64.tar.gz
  
  export JAVA_HOME="/usr/java/jre1.8.0_45/"
  
  which java:
  /usr/bin/java
  
  alias ls='ls -halF'
  
  history | grep blahblah
  
  #list disks/partitions
  lsblk
  sudo sfdisk -l -uM
  sudo fdisk -l
  df -h
  df -h --output=source,fstype,size,used,avail,pcent,target
  
  #run fsck
    #unmount!
    sudo unmount /dev/sda1
    #check if unmounted (sda1 shouldn't appear anymore)
    df -h --output=source,fstype,target
    #run fsck
    sudo -i fsck -f -v /dev/sda1
    #mount back 
    mount /dev/sda1

HOW-TO INSTALL/

  download to /tmp => extract
  install to /opt
  sudo -i ln -s /usr/bin/sublime /usr/local/bin/Sublime\ Text\ 2/sublime_text

HOW-TO DEBUG/
  
  tail -f /var/log/syslog
  tail -f /var/log/kern.log
  
  last
  
  
UBUNTU/
  
  #console font
  sudo dpkg-reconfigure console-setup

  #disable apport (crash report)
  sudo service apport stop
  sudo -i vi /etc/default/apport
    enabled=0
  
  #fsck gui
  sudo apt-get install gparted
  sudo gparted
